# 数据语句的分类

# DDL -- 数据定义语句

create, drop, alter

# DML -- 数据操作语句

insert, update, delete

# DQL -- 数据查询语句

select

# DCL -- 数据控制语句

commit, rollback (回滚), savepoint (存档)

# create table (创建表)

创建表的时候要注意约束的设定
一列可以有多个约束条件

 // 防止创建重复表单
 create table if not exists 表名(
    列名 列数据类型 [约束1 约束2][索引][注释],
    列名 列数据类型 [约束1 约束2][索引][注释],
    列名 列数据类型 [约束1 约束2][索引][注释],
    
 )comment 表的注释 (可以用=或者" "--空格) ;
 
 // comment 表的注释可以加在表的后面或者每一列之后

最好添加注释和字符设置 (UTF-8)!!

create table if not exists student(   
     sid    integer primary key auto_increment,
     name   varchar(10) unique not null,
     age    integer,
     gender varchar(3)
  ) comment ='学生表' charset = 'utf8';

# 列数据类型

integer 整型
varchar ( ) 字符型
　---> ( ) 内要填写允许的字符长度最大值

# 数据约束条件 (五大约束)

# 1. primary key 主键 : (主键约束)

一个表里面只能有一个主键

主键默认非空，默认唯一性约束，可以设置为 null , 自动从 1 开始赋值
只有主键才能设置自动增长，自动增长一定是主键，主键不一定自动增长；

# 2. FOREIGN key : 外键约束

只有 INNODB 的数据库引擎支持外键，修改 my.ini 文件设置 default-storage-engine=INNODB
外键必须与参照列的数据类型必须相同
（数值型要求长度和无符号都相同，字符串要求类型相同，长度可以不同）。

# 3. unique : 唯一约束

该列的值唯一，不能重复

# 4. not null : 非空约束

该列的值不能为空

# 5. DEFAULT : 默认值约束

当插入数据时，如果没有明确为字段 (列名) 赋值，则自动赋予默认值

# auto_increment : 自增约束 --mysql 独有约束方法

mysql 独有的约束！！！

类似 i++
默认每条自增 1 ---> 一般主键和自增一起使用

# insert into (添加数据)

# 格式一：指定列名

insert into 表名 (列名 1, 列名 2,...)
values (值 1, 值 2,...)
values (值 1, 值 2,...);

 insert into student(name, age, gender)
	values ('赵六', 20, '男'),
   	       ('钱八', 30, '女');

注:
1. 列名可以不按照顺序，但是列名和值要一一对应
2. 自增主键可以不赋值，或者留空 (设置为 null)
3. 添加多条数据 多条添加时，value () 后面加 ","

# 格式二：不指定列名

insert into 表名
values (值 1, 值 2)

注:
1. 需要严格按照建表的列名和顺序来添加值
2. 不指定列名 就要按照创建表时候的列的顺序，依次给值

    insert into student  
		values (null, '王五', 29, '男');

# update (修改)

格式
update 表名 set 列名 = 新值，列名 = 新值 where 条件
可以同时修改多个值

	update student
    set age = 14 , gender = '男'
    // 所有被 where 匹配的数据都会修改
    where sid > 7 and sid < 9;

# delete (删除行数据)

格式
delete from 表名 where 条件；

		delete
		from student
		where sid > 7;

# drop (删除整个表)

格式
drop table 表名

# select (查询)

格式 :
select 列名，列名，列名
from 表名
where 条件
[order by 列名 ASC / DESC] --> 不是必须要写
[limit 从第 n+1 几行开始，一共获取几行]; --> 不是必须要写

# 子查询

eg : where age>(),
先查询 () 中的值 以它为条件再查询 --> 类似于 "嵌套"

# 适用于一个SQL语句的条件需要另一条SQL语句的结果,
# 那么就可以使用()把结果SQL包含进来

select name, age
from student
where age >= (select age
              from student
              where name = '张根硕');

# where 查询中常用的语句

# 1. * : 查询所有列

select *
from student;

# 2. order by : 排序

默认升序排列 升序: asc 降序: desc

order by sid desc

# 3. limit : 截取

首行的序号为 0, 类似数组下标

limit 3     // 截取自首行开始的三行数据
limit 1,3   // 从第2行开始 截取三行数据

# 4. and or not

可以调整若干条件之间的关系
注意代码执行的优先级问题:
not > and > or , 使用时最好用 ( ) 提升优先顺序

# 5.= > <

# 6.in

列名中的值是否包含在 in 后面的括号中，如果是，返回查询结果
--> 相当于 or 的增强版，可以 "多选"

# 也可以使用 sid not in(1,5,6,9) 或者 (!(sid in (1,5,6,9))
select name, age, gender, sid
from student
# sid = 1 或 5 或 6 或 9
where sid in (1, 5, 6, 9);

# 7. like 模糊查询

常与 % 或 _配合使用
多是针对字符串的查询

select name, sid, age, gender
from student
     # 张% -- 张某某
     # 张_ --张某
     # % 任意长度的任意字符
     # _ 长度是1的任意字符
     # where name not like '张_'; --排除张某
     # where name not like '__'; --排除两个字的人
where name like '张_';

# 8. xor 异或

两个值不同返回 true 相同返回 false

eg: 聊天异性匹配

# 9. between ... and...

相当于 [a,b]"大于等于 a, 小于等于 b"

select sid, name, age, gender
from student
where sid between 7 and 13;

# 10. is null 为空

select sid, name, age, gender
from student
where gender is null;
