# 删除文件夹中所有文件 (利用递归)

 public static void deleteAll(File file){
        // 如果是文件, 直接删除
        if (!file.isDirectory()){
            file.delete();
            return;
        }

        // 如果是文件夹, 执行如下操作
        File[] files = file.listFiles();
        for (File f : files) {
        	   // 递归结束的出口
            if (f.isDirectory())
            	// 使用递归
                deleteAll(f);
            f.delete();
        }
        // 将删除了所有文件的该级文件夹本身也删除掉
        file.delete();
    }

# IO 流

input (读取) 输入流，指的是计算机从文件读取数据
output (写入) 输出流，指的是计算机将数据写入文件

# 流

指程序中有能力产出数据 (数据源) 或者接收数据 (接收端) 的对象
实际上流屏蔽了计算机处理数据的细节

# 流的类型

字符流：用于处理文本数据
字节流：用于处理媒体数据 (二进制文件)

# 字符流的写

 FileWriter fw = null;
        File file = new File("./testfile/1.txt");
        try {
            // 开启一个流(专门向文件内写入东西)
            // --->!!!在关闭流之前 是不会向文件写入的!!!
            fw = new FileWriter(file);
            fw.write("于印和张焯强在一起啊啊啊啊啊");
        } catch (IOException e) {
            e.printStackTrace();
            // finally ---> 无论如何都要将流关掉!!!
        } finally {
            // 把流关闭时才会写入文件 (其他时间流在内存里待着)
            try {
                // 防止fw创建失败(空指针异常)

                if (fw != null) {
                    fw.close();
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }

# 字符流的读

FileReader fr = null;
        try {
            fr = new FileReader("./testfile/1.txt");
            // 从文件读取数据

            // 定义每次读取文本的长度,将读取的内容存储到buffer数组中
            // buffer 相当于一个缓冲区
            char buffer[] = new char[5];
            // 每次读取长度判断用
            int len = 0;
            // fr.read()  --> 返回int值 (读取出来的字符串长度)
            // 每次尝试去将buffer数组读满
            // 并且返回该次读取的长度
            // 当长度不为-1的时候, 说明后续还有内容 , 所以继续循环
            // 一直到读取的长度是-1为止
            while ((len = fr.read(buffer)) != -1) {

                // 从下标 0 开始 转换几个数
                // 如果不规定起始位置和长度 ,输出会有异常
                // 内存用到哪里 哪里的内容才会被覆盖掉 不规定len 则每次都是输出五个字符
                // 第一次为"于印和张焯"
                //第二次的为"强在一起焯"
                // 参数1: 要转为字符串的char数组
                // 参数2: 从哪个下标开始转
                // 参数3: 要转换的长度
                System.out.println(new String(buffer, 0, len));
            }

        } catch (FileNotFoundException e) {
            e.printStackTrace();
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            try {
                if (fr != null) {
                    fr.close();
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
