# 数据类型

var a = 100
alert (typeof (a) ) //alert 提示框 typeof () 获取类型

# number (数字类型)

JS 中没有浮点型，都是 number 类型 (弱类型语言)

# string (字符串类型)

字符串赋值时使用 "" 包裹内容
字符串可以进行拼接 (使用 + 号)
数值和字符进行拼接时，发生隐式类型转换 (数值 --> 字符串；布尔值 --> 0/1)
注意！！从左到右的顺序 会影响拼接效果
"3" + 4 + 5 ---> "345" ; 3 + 4 + "5" ---> "75"

# boolean (bool) 布尔值

只有 true (1) , false (0) 两种
用作逻辑判断 (if 判断中，亦可遵循 "非 0 即为真")

# undefined

变量定义但未初始化赋值时返回 (空的变量)
console 数值为 NaN
typeof 为 undefined

# null

访问一个尚未存在的对象时所返回的值 (空的对象)
console 数值为 0 (表示空指针对象)
typeof 为 object

# 算数运算符

# % (取余)

取余常用来判断奇偶，提取数字各个位的值 (与 parseInt ()--> 取整 结合使用)

	//提取百位的值
   var fir = parseInt(d / 100); 
   //提取十位的值 (亦可在提取百位后,减去百位)
   var sec = parseInt(d / 10)%10;
   //提取个位的值
   var thr = d % 10

# ++/-- (自加 / 自减)

n++ 和 ++n 都是自身值加一 区别在于返回值 (赋值)

		var n = 3;
        var x = n++;// 先返回(赋值) , 再自加
        var y = ++n;//先自加 , 再返回(赋值)
        console.log(x,n);//输出 x = 3  n = 4
        console.log(y,n);//输出 y = 4  n = 4

# + - * / (加 减 乘 除)

注意有简写: += -= *= /= %=
i += 2 等价于 i = i + 2

# 比较运算符

# 不等式 <<=> >= !=

# 等式 == --> 数值相等即为相等 === --> 严格相等 (类型也要相等)

		var z = "10" == 10;//值相等即为相等
        console.log(z);//true  自动类型转换 将数字转换为字符串

        var m = "10" === 10;//值和类型完全相等才行
        console.log(m);//false

# 元素获取 操作

  oName.style.backgroundColor = "red";
  oName.className = "name"
  
  //获取或设置的标签里面的内容 -- 内容指标签内的所有子标签及文本

  var str = oName.innerHTML;
